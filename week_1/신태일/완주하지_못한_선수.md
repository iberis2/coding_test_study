function solution(participant, completion) {
  let hashMap = new Map();

  for (const name of participant) {
    let prevName = hashMap.get(name);

    hashMap.set(name, (prevName || 0) + 1);
  }

      hashMap.set(arrive, hashMap.get(arrive) - 1);
    }
  }

    if (value > 0) {
      return key;
    }
  }
}
